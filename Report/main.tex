\chapter{Introduction}

Distributed learning helps address the challenges faced by large-scale machine learning tasks by outsourcing its computations to worker nodes. Such a system is susecptible to communication bottenecks caused by end-to-end delays between the workers and the central server. Another possible issue is the presence of stragglers (nodes that are slow to respond). In this scenario, the central server does not have enough data to compute the gradient to update its model parameters. One way to mitigate this is to split the dataset into $k$ smaller batches and assign $d$ batches to each of the $n$ workers. The $i$-th worker then sends a function $f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d})$ of the gradients to the central server, where $g_{i_j}$ is the gradient computed by the worker on the $j$-th batch assigned to it. The server is able to recover the full gradient even if some of the workers are stragglers, as long as it receives enough $h_i$'s. We shall now formally define the problem of gradient coding and then discuss a communication efficient scheme to solve it.

% Introduction
\section{Problem Statement}

Consider a dataset $D = \{(x_i, y_i)\}_{i=1}^N$ where $x_i \in \mathbb{R}^l, y_i \in \mathbb{R}$. Our goal is to minimize the loss function $L(D; w) =  \sum_{i=1}^N L(w; x_i, y_i)$ where $w \in \mathbb{R}^d$ is the model parameter. One way to do this to perform:

\begin{equation}
    w^{(t+1)} = h(w^{(t)}, g^{(t)})
\end{equation}

Where $g^{(t)} = \sum_{i=1}^N \nabla L(w^{(t)}; x_i, y_i)$ and $h$ is some gradient based update. In a distributed setting, we assume $n$ workers $W_1, W_2, \ldots, W_n$ and split the dataaset $D$ into $k$ equal batches $D_1, D_2, \ldots, D_k$. Each worker is assigned $d$ batches and computes the following:

\begin{equation}
    g_i^{(t)} = \sum_{(x, y) \in D_i} \nabla L(w^{(t)}; x, y)
\end{equation}

Hence we see that $g^{(t)} = \sum_{i=1}^k g_i^{(t)}$. We assume that $s$ worker nodes are stragglers and hence we only have access to $n-s$ of the $g_i^{(t)}$'s. Say for each$i \in 1, 2, \ldots n$ we represent the datasets assigned to $W_i$ as $\{D_{i_1}, D_{i_2}, \ldots, D_{i_d}\}$. The $i$-th worker then sends a function $f_i(g_{i_1}^{(t)}, g_{i_2}^{(t)}, \ldots, g_{i_d}^{(t)})$ of the partial gradients. This paper only considers linear functions of the form: 

\begin{equation}
    f_i(g_{i_1}^{(t)}, g_{i_2}^{(t)}, \ldots, g_{i_d}^{(t)}) = \sum_{j=1}^d a_{i_j} g_{i_j}^{(t)}
\end{equation}

The partial gradient vector $g_i^{(t)} = [g_i^{(t)}(0), g_i^{(t)}(1), \ldots, g_i^{(t)}(l-1)]$. We can further generalize $f_i$ to be a linear combination of the coordinates of the partial gradients. This may require a large value of $l$ to be tolerant to stragglers or use a smaller $l$ but have less tolerance to stragglers. We shall drop the superscript $(t)$ for simplicity.

\chapter{Main Theorem}

\section{Main Result}

\begin{theorem}
    Given integers $n, k$ a triple $(d, s, m)$ where $1 \leq d \leq k$ and $m \geq 1$ is achievable if there is a distributed synchronous gradient coding scheme with $n$ workers and $k$ data batches such that:

    \begin{enumerate}
        \item Each worker is assigned $d$ data batches
        \item There are $n$ functions $f_i: \mathbb{R}^{dl} \to \mathbb{R}^{l/m}$ such that the server can recover the full gradient $g$ from any $n-s$ of the vectors $f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d})$ where $i_1, i_2, \ldots, i_d$ are the indices of the data batches assigned to worker $W_i$.
        \item Each $f_i$ is a linear combination of the coordinates of the partial gradients $g_{i_1}, \ldots, g_{i_d}$
    \end{enumerate}

    \noindent 
    and such a triple is achievable if and only if the following condition holds:

    \begin{equation}
        \frac{d}{k} \geq \frac{s+m}{n}
    \end{equation}
\end{theorem}

We shall first show that any achievable $(d, s, m)$ must satisfy the above condition and then present the paper's coding scheme for $n = k$ that satisfies $d \geq s + m$ with equality.

\section{Forward Direction}

We assume that the triple $(d, s, m)$ is achievable and show that it must satisfy the condition $\frac{d}{k} \geq \frac{s+m}{n}$. Before wew proceed, we prove the following claim:
\begin{claim}
    For each $i \in [k]$, data batch $D_i$ is assigned to at least $s + m$ workers.
\end{claim}

\begin{proof}
    Say for some $j \in [k]$ data batch $D_j$ is assigned to $a < s + m$ workers. WLOG assume these workers are $W_1, W_2, \ldots, W_a$ and suppose that first $s$ of these workers are stragglers. Based on our definition of achievability, the server should be able to recover the full gradient $g = g_1 + \ldots + g_k$ from the vectors $f_{s+1}, f_{s+2}, \ldots, f_n$.
    
    As we can clearly calculate $g_j$ using the vectors $\{ g_i : i \in [k] \setminus \{j\} \} \cup \{g_1 + \ldots + g_k\}$, we can see that $g_j$ can be obtained using the vectors $\{ g_i : i \in [k] \setminus \{j\} \} \cup \{f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d}) : s+1 \leq i \leq n\}$. Note that $D_j$ is only assigned to the first $a$ workers and $\{f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d}) : a+1 \leq i \leq n\}$ can be otained using values in the set $\{ g_i : i \in [k] \setminus \{j\} \}$. Hence $g_j$ can be obtained using only the vectors $\{ g_i : i \in [k] \setminus \{j\} \} \cup \{f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d}) : s+1 \leq i \leq a\}$. 

    Note that all $f_i$'s are linear functions of coordinates of the partial gradients and using the fact that $g_j$ is an $l$-dimensional vector, we see that the set $\{f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d}) : s+1 \leq i \leq a\}$ must contain atleast $l$ linear combinations of the coordinates of $g_j$. Since $\{f_i(g_{i_1}, g_{i_2}, \ldots, g_{i_d}) : s+1 \leq i \leq a\}$ contains elements that are $l/m$-dimensional, we see that we can have atmost $
    \frac{l}{m}(a-s)$ linear combinations of the coordinates of $g_j$. Hence we must have $\frac{l}{m}(a-s) \geq l$ which implies that $a \geq s + m$. This is a contradicton and hence the claim is proved.
\end{proof}

We shall now uuse this to prove the forward direction of the main theorem.

\begin{proof}
    From the claim we see that each data batch is assigned to atleast $s+m$ workers hence in total there must be atleast $k(s+m)$ data subsets assigned over all the workers. Since each of the $n$ workers is assigned $d$ data batches, there are a total of $nd$ data subsets assigned over all the workers. Hence we must have $nd \geq k(s+m)$ which implies that $\frac{d}{k} \geq \frac{s+m}{n}$.
\end{proof}

\begin{proof}
    \noindent We construct an explicit coding scheme that achieves $d = s + m$ with equality in the following steps:
    \begin{enumerate}
        \item Define polynomials $p_i(x)$ that encode which data subsets are assigned to each worker.
        \item Construct the encoding matrix $B$ recursively using polynomial coefficients.
        \item Show that any $(n-s)$ workers can jointly recover the full gradient vector.
    \end{enumerate}

    \paragraph{Notation}
    Let $\theta_1, \theta_2, \ldots, \theta_n$ be $n$ distinct real numbers (called ``evaluation points``). We will use these to construct polynomials and a Vandermonde-like matrix for decoding.

    \noindent \textbf{Key parameters:}
    \begin{itemize}
        \item $n$: number of workers
        \item $k = n$: number of data subsets (by assumption)
        \item $d = s + m$: computation load per worker
        \item $s$: number of stragglers tolerated
        \item $m$: communication reduction factor
        \item $l$: dimension of gradient vector (assumed divisible by $m$)
    \end{itemize}

    \paragraph{Polynomial Construction}
    For each data subset $D_i$ (where $i \in [n]$), we define a polynomial $p_i(x)$ of degree $n - d$:

    \begin{equation}
        p_i(x) = \prod_{j=1}^{n-d} (x - \theta_{i \oplus j}),
    \end{equation}

    \noindent where $\oplus$ denotes cyclic addition modulo $n$, i.e., $a \oplus b = a + b \pmod{n}$.

    \noindent The polynomial $p_i(x)$ encodes batch assignment:
    \begin{itemize}
        \item The roots of $p_i(x)$ are $\theta_{i \oplus 1}, \theta_{i \oplus 2}, \ldots, \theta_{i \oplus (n-d)}$.
        \item Data subset $D_i$ is not assigned to workers $W_{i \oplus 1}, W_{i \oplus 2}, \ldots, W_{i \oplus (n-d)}$.
        \item Therefore, $D_i$ is assigned to workers where $p_i(\theta_j) \neq 0$.
        \item This gives exactly $d = s + m$ workers per batch (by cyclic symmetry).
    \end{itemize}

    \noindent Write $p_i(x)$ in standard form:
    \begin{equation}
        p_i(x) = \sum_{j=0}^{n-d} p_{i,j} x^j,
    \end{equation}

    \noindent where $p_{i,j}$ are the coefficients. We can see that $\deg(p_i) = n - d$ and the leading coefficient $p_{i, n-d} = 1$.

    \paragraph{Recursive Polynomial Family}
    To handle the communication reduction factor $m$, we define a family of $m$ polynomials for each data subset. For each $i \in [n]$, define:

    \begin{equation}
        \begin{aligned}
            p_i^{(1)}(x) &:= p_i(x), \\
            p_i^{(u)}(x) &:= x \cdot p_i^{(u-1)}(x) - p_{i, n-d-1}^{(u-1)} \cdot p_i^{(1)}(x), \quad u = 2, 3, \ldots, m.
        \end{aligned}
    \end{equation}

    \noindent where $p_{i,j}^{(u)}$ denotes the coefficient of $x^j$ in $p_i^{(u)}(x)$. This recursive construction ensures:
    \begin{itemize}
        \item Each $p_i^{(u)}(x)$ has degree $n - d + u - 1$.
        \item The leading coefficient is always $1$: $p_{i, n-d+u-1}^{(u)} = 1$.
        \item The last $m$ coefficients align properly for reconstruction:
        $$p_{i,j}^{(u)} = 0 \quad \text{for } j \in \{n-d, n-d+1, \ldots, n-d+u-2\} \text{ when } u \geq 2.$$
        \item $p_i | p_i^{(u)}$ for all $u$, meaning:
        $$p_i^{(u)}(\theta_{i \oplus 1}) = p_i^{(u)}(\theta_{i \oplus 2}) = \cdots = p_i^{(u)}(\theta_{i \oplus (n-d)}) = 0.$$
    \end{itemize}

    \paragraph{Construction of Encoding Matrix $B$} 
    Define an $(mn) \times (n-s)$ matrix $B = (b_{ij})$ with entries:

    \begin{equation}
    b_{(i-1)m+u, j} = p_{i,j-1}^{(u)} \quad \text{for all } i \in [n], u \in [m], j \in \{1, 2, \ldots, n-s\}.
    \end{equation}

    \noindent The rows of $B$ are formed by stacking the coefficient vectors of the polynomials $p_1^{(1)}, \ldots, p_1^{(m)}, p_2^{(1)}, \ldots, p_n^{(m)}$ in order. For any $x \in \mathbb{R}$, the following matrix-vector product holds:
    \begin{equation}
    B \begin{bmatrix} 1 \\ x \\ x^2 \\ \vdots \\ x^{n-s-1} \end{bmatrix} = 
    \begin{bmatrix} 
    p_1^{(1)}(x) \\ p_1^{(2)}(x) \\ \vdots \\ p_1^{(m)}(x) \\ 
    p_2^{(1)}(x) \\ \vdots \\ 
    p_n^{(m)}(x) 
    \end{bmatrix}.
    \end{equation}

    \noindent This directly follows from the polynomial representation and definition of $B$. The last columns of $B$ form a block diagonal matrix:
    \begin{equation}
    B_{[(n-d+1):(n-s)]} = \begin{bmatrix} I_m \\ I_m \\ \vdots \\ I_m \end{bmatrix},
    \end{equation}
    \noindent where $I_m$ is the $m \times m$ identity matrix, and there are $n$ blocks of $I_m$ (one for each worker). This property is crucial for recovery, it guarantees that any $(n-s) \times (n-s)$ submatrix of the Vandermonde matrix at columns corresponding to non-straggling workers will allow us to reconstruct the full gradient.

    \paragraph{Gradient Vector Partitioning}
    We partition the gradient vector to align with the communication reduction factor. For each $v \in \{0, 1, \ldots, l/m - 1\}$ and $j \in [n]$, define an $m$-dimensional sub-vector:

    \begin{equation}
    y_v^{(j)} := \begin{bmatrix} g_j(vm) \\ g_j(vm + 1) \\ \vdots \\ g_j(vm + m - 1) \end{bmatrix} \in \mathbb{R}^m.
    \end{equation}

    \noindent For each $v$, stack these into an $(mn)$-dimensional vector:
    \begin{equation}
    z_v := \begin{bmatrix} y_v^{(1)} \\ y_v^{(2)} \\ \vdots \\ y_v^{(n)} \end{bmatrix} \in \mathbb{R}^{mn}.
    \end{equation}
    This partitioning allows for:
    \begin{enumerate}
        \item Reducing transmission from dimension $l$ to dimension $l/m$ (communication reduction by factor $m$).
        \item Applying the encoding matrix $B$ simultaneously to all $l/m$ blocks.
        \item Reconstructing each block independently using the same Vandermonde inversion.
    \end{enumerate}

    \paragraph{Encoding at Workers}
    For worker $W_i$, compute the transmitted vector:

    \begin{equation}
    f_i(g_i, g_{i \oplus 1}, \ldots, g_{i \oplus (d-1)}) := 
    \begin{bmatrix} z_0 \\ z_1 \\ \vdots \\ z_{l/m-1} \end{bmatrix} B 
    \begin{bmatrix} 1 \\ \theta_i \\ \theta_i^2 \\ \vdots \\ \theta_i^{n-s-1} \end{bmatrix}.
    \end{equation}

    \noindent This is an $(l/m)$-dimensional vector (reduced from dimension $l$). We have:
    $$p_i^{(u)}(\theta_{i \oplus 1}) = p_i^{(u)}(\theta_{i \oplus 2}) = \cdots = p_i^{(u)}(\theta_{i \oplus (n-d)}) = 0.$$

    Substituting into the encoding, the product $z_v B \begin{bmatrix} 1 & \theta_i & \cdots & \theta_i^{n-s-1} \end{bmatrix}^T$ depends only on:
    \begin{itemize}
        \item The sub-vectors $y_v^{(j)}$ for $j$ such that $p_j(\theta_i) \neq 0$, i.e., for $j \in \{i, i \oplus 1, \ldots, i \oplus (d-1)\}$.
        \item Thus, worker $W_i$ only needs access to data subsets $D_i, D_{i \oplus 1}, \ldots, D_{i \oplus (d-1)}$.
    \end{itemize}

    \paragraph{Decoding at Master Node}
     Without loss of generality, assume the non-straggling workers are $\{1, 2, \ldots, n-s\}$. Define the $(n-s) \times (n-s)$ Vandermonde matrix:

    \begin{equation}
    A := 
    \begin{bmatrix}
    1 & 1 & \cdots & 1 \\
    \theta_1 & \theta_2 & \cdots & \theta_{n-s} \\
    \theta_1^2 & \theta_2^2 & \cdots & \theta_{n-s}^2 \\
    \vdots & \vdots & \ddots & \vdots \\
    \theta_1^{n-s-1} & \theta_2^{n-s-1} & \cdots & \theta_{n-s}^{n-s-1}
    \end{bmatrix}.
    \end{equation}

    \noindent This matrix is invertible (standard property of Vandermonde matrices with distinct evaluation points). From the received vectors $f_1, f_2, \ldots, f_{n-s}$, the master obtains:

    \begin{equation}
    z_v B A \quad \text{for all } v \in \{0, 1, \ldots, l/m - 1\}.
    \end{equation}

    \noindent To recover $z_v B e_{n-d+u}$ (which corresponds to the $u$-th component of the sum gradient at position $vm + u - 1$), multiply by $A^{-1} e_{n-d+u}$:

    \begin{equation}
    z_v B e_{n-d+u} = z_v B A A^{-1} e_{n-d+u}.
    \end{equation}

    \noindent By the block diagonal structure of $B$ (equation 15), we have:
    \begin{equation}
    z_v B e_{n-d+u} = \sum_{j=1}^{n} g_j(vm + u - 1) \quad \text{for all } v \in \{0, 1, \ldots, l/m-1\}, u \in [m].
    \end{equation}

    \noindent Combining over all $v$ and $u$:
    \begin{equation}
    \left\{ z_v B e_{n-d+u} : v \in \{0, \ldots, l/m-1\}, u \in [m] \right\} = \left\{ \sum_{j=1}^n g_j(i) : i \in \{0, 1, \ldots, l-1\} \right\} = g^{(1)} + g^{(2)} + \cdots + g^{(n)}.
    \end{equation}

    \noindent Thus, the full gradient vector is recovered. We have shown that the coding scheme satisfies all required conditions, completing the achievability proof.

\end{proof}

\section{Coding Scheme}

Practical implementation requires careful handling of polynomial operations and matrix inversion.

\subsection{Computing Matrix $B$ Efficiently}

Algorithm 1 (from the original paper) provides an efficient recursive algorithm to compute the entries of $B$:

\begin{algorithm}[H]
\caption{Efficient Computation of Matrix $B$}
\label{alg:compute_B}
\begin{algorithmic}[1]
\Require{Polynomial coefficients $p_{i,j}$ for $i \in [n], j \in \{0, 1, \ldots, n-d\}$}
\Ensure{The $(mn) \times (n-s)$ matrix $B$}

\State Initialize $B$ as a zero matrix of size $(mn) \times (n-s)$

\For{$i = 1$ to $n$}
    \For{$j = 1$ to $n-d+1$}
        \State $b_{(i-1)m+1,j} \gets p_{i,j-1}$
    \EndFor
\EndFor

\For{$u = 2$ to $m$}
    \For{$i = 1$ to $n$}
        \For{$j = 2$ to $n-d+u$}
            \State $b_{(i-1)m+u,j} \gets b_{(i-1)m+u-1,j-1}$
        \EndFor
        
        \For{$j = 1$ to $n-d+1$}
            \State $b_{(i-1)m+u,j} \gets b_{(i-1)m+u,j} - b_{(i-1)m+u,n-d+1} \cdot b_{(i-1)m+1,j}$
        \EndFor
    \EndFor
\EndFor

\Return{$B$}
\end{algorithmic}
\end{algorithm}

\begin{remark}[Algorithm Complexity]
Algorithm 1 computes all $mn(n-s)$ entries of $B$ in $O(mn(n-d))$ operations. For each $(u, i)$ pair, we perform $O(n)$ operations. Since there are $mn$ pairs, total complexity is $O(mn^2)$.
\end{remark}

\subsection{Computing Polynomial Coefficients}

Before running Algorithm 1, we need to compute the base polynomial coefficients $p_{i,j}$ for all $i, j$.

\begin{procedure}[Computing $p_{i,j}$]

For each $i \in [n]$:

\begin{enumerate}
    \item Compute the roots: $\theta_{i \oplus 1}, \theta_{i \oplus 2}, \ldots, \theta_{i \oplus (n-d)}$
    
    \item Multiply out the polynomial:
    \begin{equation}
    p_i(x) = \prod_{k=1}^{n-d} (x - \theta_{i \oplus k})
    \end{equation}
    
    \item Extract coefficients: $p_{i,0}, p_{i,1}, \ldots, p_{i,n-d}$
\end{enumerate}

Each polynomial multiplication can be done in $O(n^2)$ time using FFT or direct multiplication. For all $n$ polynomials, total time is $O(n^3)$ or $O(n^2 \log n)$ with FFT.

\end{procedure}

\subsection{Vandermonde Matrix Inversion}

Computing $A^{-1}$ can be done using specialized algorithms for Vandermonde matrices:

\begin{remark}[Vandermonde Inversion Complexity]
Standard approaches:
\begin{itemize}
    \item Naive Gaussian elimination: $O((n-s)^3)$ flops
    \item Specialized Vandermonde algorithms: $O((n-s)^2)$ flops
\end{itemize}

\noindent For typical values (e.g., $n-s \approx 10-20$), even naive inversion is fast.
\end{remark}

\subsection{Choice of Evaluation Points}
The choice of evaluation points $\theta_1, \ldots, \theta_n$ affects both correctness and numerical stability.

\paragraph{Requirements}

\begin{itemize}
    \item \textbf{Correctness}: All $\theta_i$ must be distinct
    \item \textbf{Numerical Stability}: The Vandermonde matrix should be well-conditioned
\end{itemize}

\paragraph{Recommended Choice}

\noindent The original paper recommends:

\begin{equation}
\{\theta_1, \theta_2, \ldots, \theta_n\} = \begin{cases}
\{\pm(1 + i/2) : i = 0, 1, \ldots, n/2 - 1\} & \text{if } n \text{ is even} \\
\{0, \pm(1 + i/2) : i = 0, 1, \ldots, (n-1)/2 - 1\} & \text{if } n \text{ is odd}
\end{cases}
\end{equation}
